122  private void CalculateGearFactor()
123  {
124  float f = (1/(float) NoOfGears);
125  // gear factor is a normalised representation of the current speed within the current gear's range of speeds.
126  // We smooth towards the 'target' gear factor, so that revs don't instantly snap up or down when changing gear.
127  var targetGearFactor = Mathf.InverseLerp(f*m_GearNum, f*(m_GearNum + 1), Mathf.Abs(CurrentSpeed/MaxSpeed));
128  m_GearFactor = Mathf.Lerp(m_GearFactor, targetGearFactor, Time.deltaTime*5f);
129  }

259  private void SteerHelper()
260  {
261  for (int i = 0; i < 4; i++)
262  {
263  WheelHit wheelhit;
264  m_WheelColliders[i].GetGroundHit(out wheelhit);
265  if (wheelhit.normal == Vector3.zero)
266  return; // wheels arent on the ground so dont realign the rigidbody velocity

268 
269  // this if is needed to avoid gimbal lock problems that will make the car suddenly shift direction
270  if (Mathf.Abs(m_OldRotation - transform.eulerAngles.y) < 10f)
271  {
272  var turnadjust = (transform.eulerAngles.y - m_OldRotation) * m_SteerHelper;
273  Quaternion velRotation = Quaternion.AngleAxis(turnadjust, Vector3.up);
274  m_Rigidbody.velocity = velRotation * m_Rigidbody.velocity;
